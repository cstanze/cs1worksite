1    package shapes;

3    public class WorkSpace {
4        public static void main(String[] args) {

12

15           SRectangle labManual = new SRectangle(labManualWidth, labManualLength);
16           SCircle can = new SCircle(canRadius);
17           SCircle plate = new SCircle(plateDiameter / 2);
18           SSquare coaster = can.circumscribingSquare();
19   
20           double itemArea = (notebook.area() * 2) + labManual.area() + (coaster.area() * 3) + (4 * plate.area());
21           double unusedArea = table.area() - itemArea;
22   
23           System.out.println("The value of the table area used by items is " + itemArea + " inches squared.");
24           System.out.println("The value of the unused table area is " + unusedArea + " inches squared.");
25       }
