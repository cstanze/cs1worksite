1    package horde;
2    
3    import painter.SPainter;
4    import shapes.SRectangle;
5    import shapes.SCircle;
6    
7    import javax.swing.*;
8    import java.awt.*;
9    
10   public class Monster {
11       private final String name;
12       private final Color eyeColor;
13       private final Color pupilColor;
14       private double craziness;
15       private double scale;
16   
17       public Monster(String name, double craziness, int scale, Color eyeColor, Color pupilColor) {
18           this.name = name;
19           this.craziness = craziness;
20           this.scale = scale;
21           this.eyeColor = eyeColor;
22           this.pupilColor = pupilColor;
23       }
24   
25       public String toString() {
26           return "Monster: " + name + " [craziness " + craziness + "; scale " + scale + "; eyeColor " + eyeColor + "; pupilColor " + pupilColor + "]";
27       }
28   
29       public void paint(SPainter p) {
30           scale = 2.0 / Math.min(Math.max(scale, 1.0), 10.0);
31           craziness = Math.min(Math.max(craziness, 0.0), 1.0);
32           
33           // Draw the legs
34           SRectangle leg = new SRectangle(40 * scale, 20 * scale);
35           p.setColor(Color.BLACK);
36           p.mlt((leg.width() + (10 * scale)) * 5); // for centering
37           for (int i = 0; i < 10; i++) {
38               p.paint(leg);
39               p.mrt(leg.width() + (10 * scale));
40           }
41           p.mlt((leg.width() + (10 * scale)) * 5); // invariant: return
42   
43           // Draw the body
44           SRectangle body = new SRectangle(150 * scale, 300 * scale);
45           p.setRandomColor();
46           p.mfd(body.height() / 2.0);
47           p.mlt(body.width() / 15.0);
48           p.paint(body);
49           p.mbk(body.height() / 2.0);
50           p.mrt(body.width() / 15.0);
51   
52           // Draw the eyes
53           SCircle eye = new SCircle(20 * scale);
54           p.setColor(this.eyeColor);
55           p.mfd(body.height() / 4.0);
56           p.paint(eye);
57           p.mrt(body.width() / 3.0);
58           eye.expand(2 * craziness);
59           p.paint(eye);
60           p.mlt(body.width() / 3.0);
61           p.mbk(body.height() / 4.0);
62   
63           // Draw the pupils
64           SCircle pupil = new SCircle(10 * scale);
65           p.setColor(this.pupilColor);
66           p.mfd(body.height() / 4.0);
67           p.paint(pupil);
68           p.mrt(body.width() / 3.0);
69           p.paint(pupil);
70           p.mlt(body.width() / 3.0);
71           p.mbk(body.height() / 4.0);
72       }
73   }
