2     * Program to illustrate some basic character string processing functionality.

5    package stringthing;
6    
7    public class StringOps {
8        public static void main(String[] args) {
9            String date = "Wednesday, October 18, 1995";
10           String time = "8 AM";
11           String lab = "String Thing";
12   
13           int dateLength = date.length();
14           int timeLength = time.length();
15           int labLength = lab.length();
16           System.out.println("\ndateLength = " + dateLength);
17           System.out.println("timeLength = " + timeLength);
18           System.out.println("labLength = " + labLength);
19   
20           int p1 = date.indexOf(",");
21           int p2 = time.indexOf(" ");
22           int p3 = lab.indexOf("ing");
23           System.out.println("\np1 = " + p1);
24           System.out.println("p2 = " + p2);
25           System.out.println("p3 = " + p3);
26   
27           System.out.println("\ndate.substring(0,9) = " + date.substring(0,9));
28           System.out.println("time.substring(2,4) = " + time.substring(2,4));
29   
30           // could be replaced with charAt but the lab requires substring
31           System.out.println("lab.substring(7,8) = " + lab.substring(7,8));
32   
33           System.out.println("\ndate.substring(11) = " + date.substring(11));
34           System.out.println("time.substring(2) = " + time.substring(2));
35           System.out.println("lab.substring(7) = " + lab.substring(7));
36   
37           String line = date + " | " + time + " | " + lab;
38           System.out.println("\nline = " + line);
39       }
40   }
