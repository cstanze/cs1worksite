4    package stringthing;
5    public class StringThing {
6        public static void main(String[] args) {
7            // POINT A: CREATE A PRINT SOME STRINGS THAT REPRESENT NAMES
8             String singer = "Holiday, Billie";
9             String sculptor = "Claudel, Camille";
10            String painter = "Picasso, Pablo";
11            String dancer = "Zotto, Osvaldo";
12            String self = "Nieves, Jules";
13   
14           // POINT B: COMPUTE AND PRINT THE LENGTHS OF THE STRINGS, WITHOUT LABELS
15           int singerLength = singer.length();
16           int sculptorLength = sculptor.length();
17           int painterLength = painter.length();
18           int dancerLength = dancer.length();
19           int selfLength = self.length();
20   
21           System.out.println("\nString lengths ...");
22           System.out.println(singerLength);
23           System.out.println(sculptorLength);
24           System.out.println(painterLength);
25           System.out.println(dancerLength);
26           System.out.println(selfLength);
27   
28           // POINT C: COMPUTE AND PRINT THE LOCATION OF THE COMMA WITHIN EACH STRING, NO LABELS
29           int singerComma = singer.indexOf(",");
30           int sculptorComma = sculptor.indexOf(",");
31           int painterComma = painter.indexOf(",");
32           int dancerComma = dancer.indexOf(",");
33           int selfComma = self.indexOf(",");
34   
35           System.out.println("\nComma locations ...");
36           System.out.println(singerComma);
37           System.out.println(sculptorComma);
38           System.out.println(painterComma);
39           System.out.println(dancerComma);
40           System.out.println(selfComma);
41   
42           // POINT D: COMPUTE AND PRINT THE FIVE FIRST NAMES, WITH NO LABELS
43           System.out.println("\nFirst names ...");
44           System.out.println(firstName(singer));
45           System.out.println(firstName(sculptor));
46           System.out.println(firstName(painter));
47           System.out.println(firstName(dancer));
48           System.out.println(firstName(self));
49   
50           // POINT E: COMPUTE AND PRINT THE FIVE LAST NAMES, WITH NO LABELS
51           System.out.println("\nLast names ...");
52           System.out.println(lastName(singer));
53           System.out.println(lastName(sculptor));
54           System.out.println(lastName(painter));
55           System.out.println(lastName(dancer));
56           System.out.println(lastName(self));
57   
58            // POINT F: COMPUTE AND PRINT THE FIRST NAMES, AGAIN
59            System.out.println("\nFirst names, once again ...");
60            System.out.println(firstName(singer));
61            System.out.println(firstName(sculptor));
62            System.out.println(firstName(painter));
63            System.out.println(firstName(dancer));
64            System.out.println(firstName(self));
65   
66            // POINT G: COMPUTE AND PRINT THE LAST NAMES, AGAIN
67            System.out.println("\nLast names, once again ...");
68            System.out.println(lastName(singer));
69            System.out.println(lastName(sculptor));
70            System.out.println(lastName(painter));
71            System.out.println(lastName(dancer));
72            System.out.println(lastName(self));
73   
74            // POINT H: COMPUTE AND PRINT THE FULL NAMES, NATURAL STYLE
75            System.out.println("\nFull names, natural style ...");
76            System.out.println(fullName(singer));
77            System.out.println(fullName(sculptor));
78            System.out.println(fullName(painter));
79            System.out.println(fullName(dancer));
80            System.out.println(fullName(self));
81       }

83       static String firstName(String dsn) {
84           int comma = dsn.indexOf(",");
85           return dsn.substring(comma + 2);
86       }
87   
88       static String lastName(String dsn) {
89           int comma = dsn.indexOf(",");
90           return dsn.substring(0, comma);
91       }
92   
93       static String fullName(String dsn) {
94           return firstName(dsn) + " " + lastName(dsn);
95       }
96   }
