1    package npw;
2    
3    import javax.swing.*;
4    import painter.SPainter;
5    import shapes.SRectangle;
6    import java.awt.Color;
7    
8    public class PinkRectangles {
9        private void paint() {
10           SPainter p = new SPainter("Pink Rectangles", 800, 800);
11           p.setColor(Color.PINK);
12   
13           SRectangle rec = new SRectangle(100, 200); // really? only 100x200?
14           p.setPosition(p.center());
15           p.paint(rec);
16   
17           /* 
18            * Just to make this a little more interesting, let's impose even more constraints on the 
19            * problem. Currently, we must only use one SRectangle object. 
20            * Now, let's say we must use loops and must not use hard-coded positions (offset calculations are okay) or dimensions. 
21            * 
22            * Additionally, we must force the following pattern: 
23            * 1. Draw a rectangle of 100x200 at the center 
24            * 2. Rotate the rectangle 90 degrees clockwise, then draw again at a different pos so that it doesn't overlap 
25            * 3. Repeat step 2 until the rectangle has been drawn 4 times around the center 
26            */
27   
28           rec.shrink(rec.height() / 2, rec.width() / 2);
29   
30           double originalWidth = rec.width();
31           double originalHeight = rec.height();
32   
33           double xBase = p.center().getX() - (originalWidth * 2);
34           double yBase = p.center().getY() - (originalHeight * 1.75);
35           for (int i = 0; i < 4; ++i) {
36               double width = rec.width();
37               p.setPosition(p.center()); // reset position to center, makes calculations easier
38   
39               // rotation is just width <-> height swapping
40               rec.setWidth((int) rec.height());
41               rec.setHeight((int) width);
42   
43               // move to respective position
44               double xMod = -(Math.abs(i - 2)) + 2; // 0 -> 0; 1 -> 1; 2 -> 2; 3 -> 1
45               double yMod = -(Math.abs(i - 1)) + 2; // 0 -> 1; 1 -> 2; 2 -> 1; 3 -> 0
46               double locX = xBase + ((originalWidth * 2) * xMod);
47               double locY = yBase + ((originalHeight * 2) * yMod);
48   
49               p.location.setLocation(locX, locY);
50               p.paint(rec);
51           }
52       }
53   
54       public PinkRectangles() { paint(); }
55   
56       public static void main(String[] args) {
57           SwingUtilities.invokeLater(new Runnable() {
58               public void run() {
59                   new PinkRectangles();
60               }
61           });
62       }
63   }
