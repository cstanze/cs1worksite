12   public class Invention2 {
13       private void paint() {
14           int canvasWidth = getNumber("Canvas width");
15           int canvasHeight = getNumber("Canvas height");
16   
17           SPainter p = new SPainter("Invention 2", canvasWidth, canvasHeight);
18           SCircle invisibleCircle = new SCircle(100);
19           SSquare sq = new SSquare(1);
20   
21           int i = 0; // v
22           int j = 0; // h
23           while (i < canvasHeight + 1) {
24               while (j < canvasWidth + 1) {
25                   Point2D.Double pt = new Point2D.Double(j + 1, i + 1);
26                   p.setPosition(pt);
27                   p.setColor(blendedSkyColor((double) (canvasHeight - i) / canvasHeight));
28                   p.paint(sq);
29   
30                   if (p.center().distance(pt) <= invisibleCircle.radius()) {
31                       int dist = (int) (255 * ((invisibleCircle.radius() - p.center().distance(pt)) * 0.01));
32                       Color col = new Color(255, Math.abs(255 - (dist + 50)), dist / 2);
33                       p.setColor(col);
34   
35                       if (Math.random() <= 0.8) {
36                           p.setColor(blendWithSky(Math.random(), col));
37                       }
38   
39                       p.paint(sq);
40                   }
41   
42                   j++;
43               }
44               i++;
45               j = 0;
46           }
47       }
48   
49       private Color blendWithSky(double t, Color c) {
50           // c <- (0.5, 0.7, 1.0)
51           double r = c.getRed() / 255.0, g = c.getGreen() / 255.0, b = c.getBlue() / 255.0;
52           double blendedR = r * 0.5,
53                   blendedG = g * 0.7,
54                   blendedB = b * 1.0;
55   
56           return new Color((int)Math.floor(blendedR * 255), (int)Math.floor(blendedG * 255), (int)Math.floor(blendedB * 255));
57       }
58   
59       private Color blendedSkyColor(double t) {
60           // (1.0, 1.0, 1.0) <- (0.5, 0.7, 1.0)
61           double blendedR = (1.0 - t) + t * 0.5,
62                   blendedG = (1.0 - t) + t * 0.7,
63                   blendedB = (1.0 - t) + t;
64   
65           return new Color((int)Math.floor(blendedR * 255), (int)Math.floor(blendedG * 255), (int)Math.floor(blendedB * 255));
66       }

68       private static int getNumber(String prompt) {
69           String nss = JOptionPane.showInputDialog(null, prompt + ": ");
70           Scanner sc = new Scanner(nss);
71           return sc.nextInt();
72       }
