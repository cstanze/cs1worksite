5    public class CaveMap {

10           int gameWidth = sc.nextInt();

13           int caveWidth = sc.nextInt();

23       public static void drawMap(int caveWidth, int gameWidth) {
24           int mapOffset = -(gameWidth - caveWidth);
25           for (int i = mapOffset; i < (gameWidth - caveWidth) + 1; i++) {
26               int count = (-Math.abs(i)) + (gameWidth - caveWidth);
27   
28               System.out.print("*".repeat(count));
29               System.out.print(" ".repeat(caveWidth));
30               System.out.println("*".repeat(gameWidth - (count + caveWidth)));
31           }
32       }
33   }
